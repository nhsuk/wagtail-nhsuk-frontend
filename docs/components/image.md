# Images

```py
from wagtail.core.models import Page
from wagtail.core.fields import StreamField

from wagtailnhsukfrontend.blocks import ImageBlock


class HomePage(Page):

    parent_page_types = ['wagtailcore.Page']

    body = StreamField([
        ...
        ('captionable_image', ImageBlock()),
        ...
    ], use_json_field=True)
```
